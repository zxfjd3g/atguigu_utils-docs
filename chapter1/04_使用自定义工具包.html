# 4.使用自定义工具包

# 4.1.下载工具包

npm install a-utils2

# 4.2.网页中引入并使用

<script src="./node_modules/a-utils2/dist/atguigu-utils.js"></script>
<script>
  aUtils.test()
</script>

# 4.3.模块化引入并使用

// 使用ESM引入
import {test} from 'a-utils2'
test()

// 使用commonjs引入
const {test} = require('a-utils2')
test()
